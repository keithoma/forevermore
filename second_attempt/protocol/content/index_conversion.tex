dimension 1

\begin{align*}
    x \mapsto& \idx{\frac{j}{n}} := j \\
    y \mapsto& \idxi{y} := \frac{y}{n}
\end{align*}

dimension 2

\begin{align*}
    (x_1, x_2) \mapsto& \idx{\frac{j_1}{n}, \frac{j_2}{n}} := (j_1 - 1) \cdot (n - 1) + j_2
\end{align*}

dimension 3

\begin{align*}
    (x_1, x_2, x_3) \mapsto& \idx{\frac{j_1}{n}, \frac{j_2}{n}, \frac{j_3}{n}} := (j_1 - 1) \cdot (n - 1)^2 + (j_2 - 1) \cdot (n - 1) + j_3
\end{align*}
